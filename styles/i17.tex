% use \parencite for indirect citations at the end of the sentence/paragraph
% use \textcite for direct citations
\usepackage[
  backend=biber,                                            % Backend for bibtex
  url=false,                                                % Removes url keyword
  style=apa,                                                % Style in the bibliography
  citestyle=authoryear,                                     % Citation style
  maxnames=2,                                               % Maximum number in citations in text (they do et. al always)
  minnames=1,                                               % Minimum number in citation in text
  maxbibnames=99,                                           % Maximum number of authors listed in the bib
  giveninits,                                               % Only show the first name abbreviated
  uniquename=init,                                          % Only show the first name abbreviated
  dashed=false,                                             % Get duplicated entries with different years
  doi=true,                                                 % Disable Dois
  isbn=false,                                               % Disable ISBN / ISSN
  date=year                                                 % Disable months in bibliography
]{biblatex}
\renewcommand*{\nameyeardelim}{\addcomma\space}             % Adds a comma between author and year for parencite
\DeclareLanguageMapping{british}{british-apa}               % I adapted the american apa
\renewcommand*{\multinamedelim}{;\space}                    % Correct delimiter
\renewcommand*{\finalnamedelim}{\space\&\space}             % Correct delimiter
\renewrobustcmd*{\bibinitdelim}{}                           % Remove space between multiple first names

\captionsetup{labelfont=bf}                                 % Bold captions
\renewcommand*{\figureformat}{\figurename~\thefigure}       % Remove and adds figurename trailing dot

% ToC modifications
\renewcommand{\lstlistingname}{Algorithm}                   % Change Listing to Algorithm
\renewcommand{\lstlistlistingname}{List of \lstlistingname{}s}  % Change Listings  to List of Algorithms in ToC
\renewcommand{\cftchapdotsep}{1}                            % Change the seperation between the dots for chapters
\renewcommand{\cftdotsep}{1}                                % Change the speration between the dots for everything else

% Tweaks for the list of tables, figures and listings
\makeatletter
\begingroup\let\newcounter\@gobble\let\setcounter\@gobbletwo
  \globaldefs\@ne \let\c@loldepth\@ne
  \newlistof{listings}{lol}{\lstlistlistingname}
\endgroup
\let\l@lstlisting\l@listings
\makeatother

\setlength{\cftlistingsindent}{1.5em}
\renewcommand*{\cftlistingspresnum}{\lstlistingname~}
\settowidth{\cftlistingsnumwidth}{\cftlistingspresnum}
\addtolength{\cftlistingsnumwidth}{2em}                     % Tweak this for different length for lstlisting
\addtolength{\cftfignumwidth}{2em}                          % Tweak this for different length for figures
\addtolength{\cfttabnumwidth}{2em}                          % Tweak this for different length for tables
\renewcommand{\cftlistingsaftersnum}{:}

% Caption styling
\captionsetup{%
    format=plain,
    justification=centering,
    singlelinecheck=off
}

\DefineBibliographyStrings{english}{%                       % Renames Bibliography to References
    bibliography = {References},
    references = {References},
}
